int CountOpenPositions()
  {
   int count = 0;
   for(int i = PositionsTotal() - 1; i >= 0; i--)
     {
      if(posn.SelectByIndex(i))
        {
         int m_id = (int)posn.Magic();
         bool is_ifx_trade = (m_id >= i_base_magic && m_id <= i_base_magic + 50);
         
         if(posn.Symbol() == _Symbol && is_ifx_trade)
           {
            count++;
           }
        }
     }
   return count;
  }

// 🚀 WEEKEND CARRY-OVER FIX: Absolute Mathematical Midnight
int CountTodayTrades()
  {
   int count = 0;
   MqlDateTime dt;
   TimeToStruct(TimeCurrent(), dt);
   dt.hour = 0;
   dt.min = 0;
   dt.sec = 0;
   datetime start_of_day = StructToTime(dt); // True midnight
   datetime last_entry_time = 0;
   
   if(HistorySelect(start_of_day, TimeCurrent()))
     {
      for(int i = 0; i < HistoryDealsTotal(); i++)
        {
         ulong ticket = HistoryDealGetTicket(i);
         if(HistoryDealGetInteger(ticket, DEAL_ENTRY) == DEAL_ENTRY_IN &&
            HistoryDealGetString(ticket, DEAL_SYMBOL) == _Symbol)
           {
            int m_id = (int)HistoryDealGetInteger(ticket, DEAL_MAGIC);
            if(m_id >= i_base_magic && m_id <= i_base_magic + 50)
              {
               datetime deal_time = (datetime)HistoryDealGetInteger(ticket, DEAL_TIME);
               // Only count if more than 5 seconds from the last counted entry
               // This deduplicates the TP1+TP2 split-order pair
               if(deal_time - last_entry_time > 5)
                 {
                  count++;
                  last_entry_time = deal_time;
                 }
              }
           }
        }
     }
   return count;
  }

double ParseValue(string txt, string keyword, double fallback)
  {
   string up_txt = txt;
   StringToUpper(up_txt);
   string up_kw = keyword;
   StringToUpper(up_kw);
   int pos = StringFind(up_txt, up_kw);
   
   if(pos < 0)
      return fallback;
   
   int start = pos + StringLen(up_kw);
   
   // 🛑 JSON NESTED ARRAY FIX: If we are searching for Levels (Target 1, Target 2, etc),
   // we must seek forward to the actual "PRICE POINT" key in the JSON object
   // before extracting numbers to avoid capturing numbers inside the condition notes.
   if(StringFind(up_kw, "TARGET 1") >= 0 || StringFind(up_kw, "TARGET 2") >= 0 || StringFind(up_kw, "INVALIDATION") >= 0 || StringFind(up_kw, "COUNTER") >= 0)
     {
      int price_pos = StringFind(up_txt, "PRICE POINT", start);
      if(price_pos > start)
        {
         start = price_pos + StringLen("PRICE POINT");
        }
     }
   
   string num_str = "";
   bool found_digit = false;
   
   for(int i = start; i < StringLen(up_txt); i++)
     {
      ushort c = StringGetCharacter(up_txt, i);
      if(c >= '0' && c <= '9')
        {
         num_str += ShortToString(c);
         found_digit = true;
        }
      else
          if(c == '.' || c == ',')
           {
            // Only allow one decimal point
            if(StringFind(num_str, ".") < 0)
               num_str += ".";
            found_digit = true;
           }
         else
            if(c == ',')
              {
               // Ignore commas (thousands separators)
               continue;
              }
         else
           {
            if(found_digit)
               break;
           }
     }
   return (num_str != "") ? StringToDouble(num_str) : fallback;
  }

        local_atr = atr[0];
    } else {
        // 🛡️ DYNAMIC VOLATILITY FALLBACK: Mathematically evaluate active chart structure
        double h_arr[], l_arr[];
        if(CopyHigh(_Symbol, i_tf_enforce, 0, 100, h_arr) > 0 && CopyLow(_Symbol, i_tf_enforce, 0, 100, l_arr) > 0) {
            double h_max = h_arr[ArrayMaximum(h_arr)];
            double l_min = l_arr[ArrayMinimum(l_arr)];
            local_atr = h_max - l_min;
        } else {
            local_atr = 500 * _Point;
        }
    }
   
    double opt_pct = GetOptimalATRPercentage(_Symbol);
    double zone_thickness = local_atr * (opt_pct / 100.0);
    zoneHigh = NormalizeDouble(sys_pivot + (zone_thickness / 2), _Digits);
    zoneLow  = NormalizeDouble(sys_pivot - (zone_thickness / 2), _Digits);
    calc_inval = (sys_bias == "Long") ? zoneLow : zoneHigh;
    ui_needs_redraw = true;
}

string ParseString(string txt, string keyword)
  {
   int pos = StringFind(txt, keyword);
   if(pos < 0) return "";
   
   int start = pos + StringLen(keyword);
   int colon = StringFind(txt, ":", start);
   if(colon >= 0) start = colon + 1;
   
   int quote1 = StringFind(txt, "\"", start);
   if(quote1 < 0) return "";
   int quote2 = StringFind(txt, "\"", quote1 + 1);
   if(quote2 < 0) return "";
   
   return StringSubstr(txt, quote1 + 1, quote2 - quote1 - 1);
  }

void ParseAIText(string up_txt) {
    if(StringLen(up_txt) > 5)
     {
      string old_bias = sys_bias;
      double old_pivot = sys_pivot;

      string extracted_bias = ParseString(up_txt, "\"BIAS\"");
      if(extracted_bias != "") {
         if(StringFind(extracted_bias, "BULLISH") >= 0 || StringFind(extracted_bias, "LONG") >= 0)
            sys_bias = "Long";
         else if(StringFind(extracted_bias, "BEARISH") >= 0 || StringFind(extracted_bias, "SHORT") >= 0)
            sys_bias = "Short";
         else
            sys_bias = "Neutral";
      } else {
         if(i_manual_bias == BIAS_LONG) sys_bias = "Long";
         else if(i_manual_bias == BIAS_SHORT) sys_bias = "Short";
         else sys_bias = "Neutral";
      }
       
      // 🛑 JSON STRICT KEY FIX: Search explicitly for the JSON Keys instead of generic words
      // Otherwise the word 'pivot' inside the LLM reasoning text block will trigger a premature capture of 0.0
      sys_pivot = ParseValue(up_txt, "ENTRY PIVOT ZONE", i_manual_pivot);
      sys_tp1   = ParseValue(up_txt, "TARGET 1", i_manual_tp1);
      sys_tp2   = ParseValue(up_txt, "TARGET 2", i_manual_tp2);
      sys_confidence = (int)ParseValue(up_txt, "CONFIDENCE SCORE", 70);
       
      if(sys_tp1 == 0.0)
         sys_tp1 = ParseValue(up_txt, "TP1", i_manual_tp1);
      if(sys_tp2 == 0.0)
         sys_tp2 = ParseValue(up_txt, "TP2", i_manual_tp2);

      sys_tp1 = NormalizeDouble(sys_tp1, _Digits);
      sys_tp2 = NormalizeDouble(sys_tp2, _Digits);

      // 🚨 BIAS-FLIP AUTO-CLOSE LOGIC
      if(i_exit_on_flip && old_bias != sys_bias && sys_bias != "Neutral" && CountOpenPositions() > 0)
        {
         bool closed = false;
         for(int i = PositionsTotal() - 1; i >= 0; i--)
           {
            if(posn.SelectByIndex(i) && posn.Symbol() == _Symbol && (int)posn.Magic() >= i_base_magic && (int)posn.Magic() <= i_base_magic + 50)
              {
               ENUM_POSITION_TYPE p_type = posn.PositionType();
               if((p_type == POSITION_TYPE_BUY && sys_bias == "Short") || (p_type == POSITION_TYPE_SELL && sys_bias == "Long"))
                 {
                  if(trade.PositionClose(posn.Ticket())) closed = true;
                 }
              }
           }
         if(closed)
            SendDiscordAlert(_Symbol + " | 🛑 BIAS INVALIDATED", "The IFX Scanner has detected a trend reversal. Current position closed to protect capital.", 15548997, 0, 0, 0, 0, 0);
        }

      // 🧠 EXTRACT AI TECHNICAL REASONING
      sys_reasoning = ParseString(up_txt, "\"STRUCTURAL_REASONING\"");
      if(sys_reasoning == "") sys_reasoning = ParseString(up_txt, "REASONING");

      if((old_bias != sys_bias || old_pivot != sys_pivot) && sys_pivot > 0.0) {
          Print("🚀 NEW VERTEX AI SIGNAL INGESTED: ", sys_bias, " @ ", sys_pivot);
          ResetStateForNewSignal();
      }
     }
   else
     {
      if(i_manual_bias == BIAS_LONG)
         sys_bias = "Long";
      else
         if(i_manual_bias == BIAS_SHORT)
            sys_bias = "Short";
         else
            sys_bias = "Neutral";
       
      sys_pivot = i_manual_pivot;
      sys_tp1   = NormalizeDouble(i_manual_tp1, _Digits);
      sys_tp2   = NormalizeDouble(i_manual_tp2, _Digits);
     }
}

void GetActiveStructure(ENUM_TIMEFRAMES tf, int len, double &out_ph, double &out_pl, datetime &out_t_ph, datetime &out_t_pl)
  {
   double h[], l[], c[];
   datetime t[];
   ArraySetAsSeries(h, true);
   ArraySetAsSeries(l, true);
   ArraySetAsSeries(c, true);
   ArraySetAsSeries(t, true);
   
   int lookback = i_smc_lookback;
   
   int copied_h = CopyHigh(_Symbol, tf, 0, lookback, h);
   int copied_l = CopyLow(_Symbol, tf, 0, lookback, l);
   int copied_c = CopyClose(_Symbol, tf, 0, lookback, c);
   int copied_t = CopyTime(_Symbol, tf, 0, lookback, t);
   
   if(copied_h <= len * 2 || copied_l <= len * 2 || copied_c <= len * 2 || copied_t <= len * 2)
     {
      out_ph = EMPTY_VALUE;
      out_pl = EMPTY_VALUE;
      out_t_ph = 0;
      out_t_pl = 0;
      return;
     }

   int alb = (int)MathMin(copied_h, MathMin(copied_l, MathMin(copied_c, copied_t)));
   
   double active_ph = EMPTY_VALUE, active_pl = EMPTY_VALUE;
   datetime t_act_ph = 0, t_act_pl = 0;

   // 🚀 PROXIMITY SEARCH: Walk backwards from newest to oldest
   // We stop the moment we find the NEAREST qualified structural floor/ceiling.
   for(int i = len; i < alb - len; i++)
     {
      bool isPH = true;
      bool isPL = true;
     
      // Check fractal qualification (Must be higher/lower than 'len' neighbors on both sides)
      for(int j = i - len; j <= i + len; j++)
        {
         if(i == j) continue;
         if(h[j] > h[i]) isPH = false;
         if(l[j] < l[i]) isPL = false;
        }

      // Capture the NEAREST High
      if(isPH && active_ph == EMPTY_VALUE)
        {
         active_ph = h[i];
         t_act_ph = t[i];
        }
       
      // Capture the NEAREST Low
      if(isPL && active_pl == EMPTY_VALUE)
        {
         active_pl = l[i];
         t_act_pl = t[i];
        }

      // 🎯 EXIT STRATEGY: If we have found both nearest anchors, stop searching immediately.
      if(active_ph != EMPTY_VALUE && active_pl != EMPTY_VALUE)
         break;
     }
     
   out_ph = active_ph;
   out_pl = active_pl;
   out_t_ph = t_act_ph;
   out_t_pl = t_act_pl;
  }

// Returns the broken pivot price, or 0.0 if no break detected.
// This value is used directly as the spam-guard anchor.
double CheckLiveBOS(string type, int len)
  {
   double h[], l[], c[];
   if(CopyHigh(_Symbol, i_tf_enforce, 0, 100, h) < 100) return 0.0;
   if(CopyLow(_Symbol, i_tf_enforce, 0, 100, l) < 100) return 0.0;
   if(CopyClose(_Symbol, i_tf_enforce, 0, 100, c) < 100) return 0.0;
   
   ArraySetAsSeries(h, true); ArraySetAsSeries(l, true); ArraySetAsSeries(c, true);
   
   for(int i = len + 1; i < 50; i++)
     {
      if(type == "Short")
        {
         bool isPL = true;
         for(int j = i - len; j <= i + len; j++)
           {
            if(i == j) continue;
            if(l[j] <= l[i]) { isPL = false; break; }
           }
         if(isPL)
           {
            if(c[1] < l[i] && c[2] >= l[i])
               return l[i]; // Return the exact broken pivot low price
            return 0.0;
           }
        }
      else if(type == "Long")
        {
         bool isPH = true;
         for(int j = i - len; j <= i + len; j++)
           {
            if(i == j) continue;
            if(h[j] >= h[i]) { isPH = false; break; }
           }
         if(isPH)
           {
            if(c[1] > h[i] && c[2] <= h[i])
               return h[i]; // Return the exact broken pivot high price
            return 0.0;
           }
        }
     }
   return 0.0;
  }

int OnInit()
  {
   if(!TerminalInfoInteger(TERMINAL_TRADE_ALLOWED) || !TerminalInfoInteger(TERMINAL_CONNECTED))
      Print("WARNING: Terminal not connected or Algo Trading disabled. WebRequests may fail.");

   symb.Name(_Symbol);
   if(!symb.Refresh())
      return(INIT_FAILED);
   
   sys_magic = i_base_magic + i_tf_enforce;
   trade.SetExpertMagicNumber(sys_magic);
   
   trade.SetDeviationInPoints(50);
   
   uint filling = (uint)SymbolInfoInteger(_Symbol, SYMBOL_FILLING_MODE);
   if((filling & SYMBOL_FILLING_FOK) != 0)
      trade.SetTypeFilling(ORDER_FILLING_FOK);
   else
      if((filling & SYMBOL_FILLING_IOC) != 0)
         trade.SetTypeFilling(ORDER_FILLING_IOC);
      else
         trade.SetTypeFilling(ORDER_FILLING_RETURN);

   global_atr_handle = iATR(_Symbol, PERIOD_D1, 14);
   if(global_atr_handle == INVALID_HANDLE)
      return(INIT_FAILED);

   // 🧠 PERSISTENT STATE RECOVERY
   // Load mathematical matrix strictly from shared disk to survive timeframe changes
   string payload_file = "IFX_Matrix_" + _Symbol + ".json";
   int handle = FileOpen(payload_file, FILE_READ|FILE_TXT|FILE_COMMON);
   bool recovered = false;
   
   if(handle != INVALID_HANDLE)
     {
      string json_data = "";
      while(!FileIsEnding(handle))
        {
         json_data += FileReadString(handle);
        }
      FileClose(handle);
     
      if(StringLen(json_data) > 10)
        {
         string stored_txt = json_data;
         StringToUpper(stored_txt);
         ParseAIText(stored_txt);
         recovered = true;
         Print("✅ [EXECUTOR] Ghost Memory Restored! Persistent Matrix recovered successfully.");
        }
     }
     
   if(!recovered)
     {
      string up_txt = i_ai_text;
      StringToUpper(up_txt);
      ParseAIText(up_txt);
     }

   daily_trades_count = CountTodayTrades();
   MqlDateTime dt;
   TimeToStruct(TimeCurrent(), dt);
   last_trade_day = dt.day;

   // 🛑 MT5 64-BIT TRUNCATION BUG FIX:
   // MT5 GlobalVariables only store 53-bit doubles. We MUST split the 64-bit ChartID
   // into two 32-bit halves to safely transmit it globally without corrupting the value!
   long exec_id = ChartID();
   uint id_hi = (uint)(exec_id >> 32);
   uint id_lo = (uint)(exec_id & 0xFFFFFFFF);
   GlobalVariableSet("IFX_HI_" + _Symbol, (double)id_hi);
   GlobalVariableSet("IFX_LO_" + _Symbol, (double)id_lo);

   EventSetTimer(1);
   return(INIT_SUCCEEDED);
  }

   ObjectsDeleteAll(0, "LDG_");
   ChartRedraw();
  }

  {

void ManageOpenPositions()
  {
   double total_open_lots = 0.0;
   bool tp1_exists = false;
   bool tp2_exists = false;

   for(int i = PositionsTotal() - 1; i >= 0; i--)
     {
      if(posn.SelectByIndex(i) && posn.Symbol() == _Symbol)
        {
         int m_id = (int)posn.Magic();
         if(m_id >= i_base_magic && m_id <= i_base_magic + 50)
           {
            double tp = posn.TakeProfit();
            if(sys_tp1 > 0.0 && MathAbs(tp - sys_tp1) < _Point * 10)
               tp1_exists = true;
            if(sys_tp2 > 0.0 && MathAbs(tp - sys_tp2) < _Point * 10)
               tp2_exists = true;
            total_open_lots += posn.Volume();
           }
        }
     }

   if(tp2_exists && !tp1_exists && i_usePartial && sys_tp1 > 0.0 && !is_partial_taken)
     {
      if(i_notify_on_tp)
         SendDiscordAlert(_Symbol + " | 🎯 PROFIT TARGET REACHED", "Institutional TP1 successfully executed. Position risk removed.", 5763719, 0, 0, 0, 0, 0);
      is_partial_taken = true;
     }

   bool trigger_be = false;
   if(i_be_after_tp1 && is_partial_taken)
      trigger_be = true;

   if(i_useBE && !trigger_be && tp2_exists)
     {
      double ref_entry = 0;
      long ref_type = -1;
      datetime ref_time = 0;
       
      for(int i = PositionsTotal() - 1; i >= 0; i--)
        {
         if(posn.SelectByIndex(i) && posn.Symbol() == _Symbol && (int)posn.Magic() >= i_base_magic && (int)posn.Magic() <= i_base_magic + 50)
           {
            ref_entry = posn.PriceOpen();
            ref_type = posn.PositionType();
            ref_time = posn.Time();
            break;
           }
        }

      if(ref_type != -1)
        {
         double m10_close[];
         ArraySetAsSeries(m10_close, true);
         
         if(CopyClose(_Symbol, i_be_tf, 0, 1, m10_close) > 0)
           {
            double be_ph, be_pl;
            datetime t_ph, t_pl;
            GetActiveStructure(i_be_tf, i_pivotLen, be_ph, be_pl, t_ph, t_pl);

            if(ref_type == POSITION_TYPE_BUY && be_ph != EMPTY_VALUE && be_ph > ref_entry && m10_close[0] > ref_entry && t_ph > ref_time)
               trigger_be = true;
            if(ref_type == POSITION_TYPE_SELL && be_pl != EMPTY_VALUE && be_pl < ref_entry && m10_close[0] < ref_entry && t_pl > ref_time)
               trigger_be = true;
           }
        }
     }

   if(trigger_be)
      is_be_secured = true;

   for(int i = PositionsTotal() - 1; i >= 0; i--)
     {
      if(posn.SelectByIndex(i) && posn.Symbol() == _Symbol)
        {
         int m_id = (int)posn.Magic();
         if(m_id >= i_base_magic && m_id <= i_base_magic + 50)
           {
            double entry = posn.PriceOpen();
            long type = posn.PositionType();
            double sl = posn.StopLoss();
            double tp = posn.TakeProfit();
               
            live_sl = sl;
               
            // AutoRR logic moved to entry only to prevent global variable overwriting.

            if(is_be_secured)
              {
               double new_sl = NormalizeDouble(entry, _Digits);
               if(MathAbs(sl - new_sl) > _Point * 2)
                 {
                  long stop_level_points = SymbolInfoInteger(_Symbol, SYMBOL_TRADE_STOPS_LEVEL);
                  long freeze_level_points = SymbolInfoInteger(_Symbol, SYMBOL_TRADE_FREEZE_LEVEL);
                  long spread_points = SymbolInfoInteger(_Symbol, SYMBOL_SPREAD);
                  long min_points = MathMax(MathMax(stop_level_points, freeze_level_points), MathMax(spread_points, 10));
                  double required_buffer = min_points * _Point;
                  double current_price = (type == POSITION_TYPE_BUY) ? symb.Bid() : symb.Ask();
                       
                  if(MathAbs(current_price - new_sl) > required_buffer && (tp == 0 || MathAbs(current_price - tp) > required_buffer))
                    {
                     trade.PositionModify(posn.Ticket(), new_sl, tp);
                     if(!is_be_notified)
                       {
                        SendDiscordAlert(_Symbol + " | 🛡️ RISK REMOVED (Break-Even)", "Trade protection active. Stop Loss has been moved to your Entry Price.", 3447003, entry, new_sl, tp, 0, posn.Volume());
                        is_be_notified = true;
                       }
                    }
                 }
              }
           }
        }
     }
   if(CountOpenPositions() > 0)
      live_lots = total_open_lots;
  }

//+------------------------------------------------------------------+
void CheckTradeExits()
  {
   if(HistorySelect(TimeCurrent() - 86400, TimeCurrent()))
     {
      for(int j = HistoryDealsTotal() - 1; j >= 0; j--)
        {
         ulong ticket = HistoryDealGetTicket(j);
         if(ticket == last_loss_ticket)
            break;
           
         if(HistoryDealGetInteger(ticket, DEAL_ENTRY) == DEAL_ENTRY_OUT && HistoryDealGetString(ticket, DEAL_SYMBOL) == _Symbol)
           {
            int m_id = (int)HistoryDealGetInteger(ticket, DEAL_MAGIC);
            if(m_id >= i_base_magic && m_id <= i_base_magic + 50)
              {
               double pnl = HistoryDealGetDouble(ticket, DEAL_PROFIT);
               last_loss_ticket = ticket; // Track this exit to avoid double-notifying
               
               // 1. ⚖️ BREAK-EVEN EXIT (Profit near 0)
               if(MathAbs(pnl) < 1.0)
                 {
                  SendDiscordAlert(_Symbol + " | ⚖️ BREAK-EVEN HIT (Exit)", "Position closed at entry price. No loss incurred.", 10855845, 0, 0, 0, 0, 0);
                 }
               // 2. 🛡️ REAL LOSS (Negative)
               else if(pnl < 0.0)
                 {
                  if(i_useDeadSL)
                    {
                     isDead_SL = true;
                     dead_SL_time = (datetime)HistoryDealGetInteger(ticket, DEAL_TIME);
                    }
                  if(i_notify_on_sl)
                     SendDiscordAlert(_Symbol + " | 🛡️ RISK PROTECTION TRIGGERED", "Structural invalidation reached. Trade closed for protection.", 15548997, 0, 0, 0, 0, 0);
                 }
               break;
              }
           }
        }
     }
  }

//+------------------------------------------------------------------+
void CalculateAutoRR(string dir, double sl)
  {
   if(!i_useAutoRR)
      return;
   double entry = (dir == "Long") ? symb.Ask() : symb.Bid();
   double risk = MathAbs(entry - sl);
   if(dir == "Long")
     {
      sys_tp1 = NormalizeDouble(entry + (risk * i_autoRR1), _Digits);
      sys_tp2 = NormalizeDouble(entry + (risk * i_autoRR2), _Digits);
     }
   else
     {
      sys_tp1 = NormalizeDouble(entry - (risk * i_autoRR1), _Digits);
      sys_tp2 = NormalizeDouble(entry - (risk * i_autoRR2), _Digits);
     }
  }

//+------------------------------------------------------------------+
double CalculateLots(double sl_price)
  {
   double risk_usd = acct.Equity() * (i_riskPct / 100.0);
   double tick_size = symb.TickSize();
   
   if(tick_size <= 0.0)
      return 0.0;
   
   double tick_value = symb.TickValue();
   double current_price = (sys_bias == "Long") ? symb.Ask() : symb.Bid();
   double sl_dist_points = MathAbs(current_price - sl_price) / tick_size;
   
   if (sl_dist_points == 0 || tick_value == 0)
      return 0.0;
   
   double lots = risk_usd / (sl_dist_points * tick_value);
   double min_lot = symb.LotsMin();
   double max_lot = symb.LotsMax();
   double step_lot = symb.LotsStep();
   
   if(step_lot <= 0.0)
      step_lot = 0.01;
   
   lots = MathFloor((lots + 1e-8) / step_lot) * step_lot;
   
   if(lots < min_lot)
      return i_strictRisk ? 0.0 : min_lot;
   if(lots > max_lot)
      return max_lot;
   
   int lot_digits = (int)MathMax(0, MathCeil(-MathLog10(step_lot)));
   return NormalizeDouble(lots, lot_digits);
  }

bool DrawSMCStructure()
  {
   DrawStructure();
   
   int lookback = i_smc_lookback;
   double h[], l[], c[];
   datetime t[];
   
   int copied_h = CopyHigh(_Symbol, i_tf_enforce, 0, lookback, h);
   int copied_l = CopyLow(_Symbol, i_tf_enforce, 0, lookback, l);
   int copied_c = CopyClose(_Symbol, i_tf_enforce, 0, lookback, c);
   int copied_t = CopyTime(_Symbol, i_tf_enforce, 0, lookback, t);
   
   if(copied_h <= 10 || copied_l <= 10 || copied_c <= 10 || copied_t <= 10)
      return false; // Force retry on next tick if MT5 is still downloading history
     
   ObjectsDeleteAll(0, "IFX_SMC_");
   
   int actual_lookback = (int)MathMin(copied_h, MathMin(copied_l, MathMin(copied_c, copied_t)));
   if(actual_lookback < lookback)
      lookback = actual_lookback;

   ArraySetAsSeries(h, true);
   ArraySetAsSeries(l, true);
   ArraySetAsSeries(c, true);
   ArraySetAsSeries(t, true);

   int pLen = i_pivotLen;
   int ph_bos_count = 0;
   int pl_bos_count = 0;

   for(int i = pLen; i < lookback - pLen; i++)
     {
      bool isPH = true;
      for(int j = i - pLen; j <= i + pLen; j++)
        {
         if(i == j)
            continue;
         if(h[j] >= h[i])
           {
            isPH = false;
            break;
           }
        }
       
      if(isPH)
        {
         for(int k = i - 1; k >= 0; k--)
           {
            if(c[k] > h[i])
              {
               string name = "IFX_SMC_PH_" + IntegerToString(t[i]);
               DrawTrendLine(name, t[i], h[i], t[k], h[i], clrMediumSeaGreen, STYLE_DASH);
               DrawText(name+"_txt", "BOS", t[k+(i-k)/2], h[i], clrMediumSeaGreen, ANCHOR_LOWER);
               ph_bos_count++;
               break;
              }
           }
        }

      bool isPL = true;
      for(int j = i - pLen; j <= i + pLen; j++)
        {
         if(i == j)
            continue;
         if(l[j] <= l[i])
           {
            isPL = false;
            break;
           }
        }
       
      if(isPL)
        {
         for(int k = i - 1; k >= 0; k--)
           {
            if(c[k] < l[i])
              {
               string name = "IFX_SMC_PL_" + IntegerToString(t[i]);
               DrawTrendLine(name, t[i], l[i], t[k], l[i], clrCrimson, STYLE_DASH);
               DrawText(name+"_txt", "BOS", t[k+(i-k)/2], l[i], clrCrimson, ANCHOR_UPPER);
               pl_bos_count++;
               break;
              }
           }
        }
     }
     
   return true;
  }

      double net_prof = 0.0;
      int printed = 0;
       
      for(int i = current_deals-1; i >= 0 && printed < 4; i--)
        {
         ulong ticket = HistoryDealGetTicket(i);
         if(HistoryDealGetString(ticket, DEAL_SYMBOL) == _Symbol && HistoryDealGetInteger(ticket, DEAL_ENTRY) == DEAL_ENTRY_OUT)
           {
            int m_id = (int)HistoryDealGetInteger(ticket, DEAL_MAGIC);
            if(m_id >= i_base_magic && m_id <= i_base_magic + 50)
              {
               double pnl = HistoryDealGetDouble(ticket, DEAL_PROFIT);
               net_prof += pnl;
                   
               string dir = (HistoryDealGetInteger(ticket, DEAL_TYPE) == DEAL_TYPE_BUY) ? "Buy" : "Sell";
               string pnl_str = (pnl >= 0 ? "+" : "-") + DoubleToString(MathAbs(pnl), 2);
               color pnl_bg = (pnl >= 0) ? C'40,167,69' : C'220,53,69';
                   
               int r = printed + 1;
               int y_pos = start_y + (r*row_h);
                   
               DrawTableCornerCell("lg_r"+IntegerToString(r)+"_1", "#"+IntegerToString(r), x_trade, y_pos, w_trade, row_h, c_gray, clrWhite);
               DrawTableCornerCell("lg_r"+IntegerToString(r)+"_2", dir, x_dir, y_pos, w_dir, row_h, c_gray, clrWhite);
               DrawTableCornerCell("lg_r"+IntegerToString(r)+"_3", DoubleToString(HistoryDealGetDouble(ticket, DEAL_PRICE), _Digits), x_entry, y_pos, w_entry, row_h, c_gray, clrWhite);
               DrawTableCornerCell("lg_r"+IntegerToString(r)+"_4", pnl_str, x_pnl, y_pos, w_pnl, row_h, pnl_bg, clrWhite);
               
               printed++;
              }
           }
        }
       
      while(printed < 4)
        {
         int r = printed + 1;
         int y_pos = start_y + (r*row_h);
         DrawTableCornerCell("lg_r"+IntegerToString(r)+"_1", "-", x_trade, y_pos, w_trade, row_h, c_gray, clrWhite);
         DrawTableCornerCell("lg_r"+IntegerToString(r)+"_2", "-", x_dir, y_pos, w_dir, row_h, c_gray, clrWhite);
         DrawTableCornerCell("lg_r"+IntegerToString(r)+"_3", "-", x_entry, y_pos, w_entry, row_h, c_gray, clrWhite);
         DrawTableCornerCell("lg_r"+IntegerToString(r)+"_4", "-", x_pnl, y_pos, w_pnl, row_h, c_gray, clrWhite);
         printed++;
        }

      int y_tot = start_y + (5*row_h);
      int w_total_label = w_trade + w_dir + w_entry;
      DrawTableCornerCell("lg_tot1", "TOTAL NET:", x_entry, y_tot, w_total_label, row_h, c_dark, clrWhite);
      DrawTableCornerCell("lg_tot2", (net_prof>=0?"+":"-")+DoubleToString(MathAbs(net_prof), 2), x_pnl, y_tot, w_pnl, row_h, (net_prof>=0)?c_green:c_red, clrWhite);
     }
  }

void SendDailyReport()
  {
   int wins = 0, losses = 0, break_evens = 0;
   double net_pnl = 0.0;
   
   if(HistorySelect(iTime(_Symbol, PERIOD_D1, 0), TimeCurrent()))
     {
      for(int i = 0; i < HistoryDealsTotal(); i++)
        {
         ulong ticket = HistoryDealGetTicket(i);
         if(HistoryDealGetString(ticket, DEAL_SYMBOL) == _Symbol && HistoryDealGetInteger(ticket, DEAL_ENTRY) == DEAL_ENTRY_OUT)
           {
            int m_id = (int)HistoryDealGetInteger(ticket, DEAL_MAGIC);
            if(m_id >= i_base_magic && m_id <= i_base_magic + 50)
              {
               double pnl = HistoryDealGetDouble(ticket, DEAL_PROFIT) + HistoryDealGetDouble(ticket, DEAL_COMMISSION) + HistoryDealGetDouble(ticket, DEAL_SWAP);
               double be_threshold = 2.0;
               
               if(pnl > be_threshold)
                  wins++;
               else
                  if (pnl < -be_threshold)
                     losses++;
                  else
                     break_evens++;
                     
               net_pnl += pnl;
              }
           }
        }
     }
     
   string sign = (net_pnl >= 0) ? "+$" : "-$";
   string desc = "Total Trades: " + IntegerToString(wins + losses + break_evens) + "\\nWins: " + IntegerToString(wins) + "\\nLosses: " + IntegerToString(losses) + "\\nBreak-Evens: " + IntegerToString(break_evens) + "\\nNet PnL: " + sign + DoubleToString(MathAbs(net_pnl), 2);
   if(i_notify_daily)
      SendDiscordAlert("📊 DAILY TRADE REPORT", desc, (net_pnl >= 0) ? 5763719 : 16711680, 0, 0, 0, 0, 0);
  }

void SendDiscordAlert(string title, string description, int embed_color, double pivot=0, double sl=0, double tp1=0, double tp2=0, double lots=0)
  {
   // --- MQL5 VALIDATOR SAFETY CHECK ---
   // If the user (or the validator) has disabled Discord, exit immediately.
   if(!i_enable_discord)
      return;

   if(i_discord_url == "" || i_discord_url == "https://discord.com/api/webhooks/YOUR_WEBHOOK_ID_HERE")
     {
      Print("❌ DISCORD ERROR: Webhook URL is empty or using the default placeholder.");
      return;
     }
   
   StringReplace(title, "\"", "'");
   StringReplace(description, "\\", "\\\\"); // Escape backslashes first
   StringReplace(description, "\"", "'");
   StringReplace(description, "\n", "\\n");
   char post[], result[];
   string result_headers;
   
   string req_headers = "Content-Type: application/json\r\n";
   string bridge_url = "http://127.0.0.1:8000/discord";
   string json = "{" +
                 "\"url\": \"" + i_discord_url + "\", " +
                 "\"title\": \"" + title + "\", " +
                 "\"description\": \"" + description + "\", " +
                 "\"color\": " + IntegerToString(embed_color) + ", " +
                 "\"asset\": \"" + _Symbol + "\", " +
                 "\"entry\": \"" + DoubleToString(pivot, _Digits) + "\", " +
                 "\"sl\": \"" + DoubleToString(sl, _Digits) + "\", " +
                 "\"tp1\": \"" + DoubleToString(tp1, _Digits) + "\", " +
                 "\"tp2\": \"" + DoubleToString(tp2, _Digits) + "\", " +
                 "\"calc_vol\": \"" + DoubleToString(lots, 2) + "\", " +
                 "\"challenge_size\": \"$1,000.00\"" +
                 "}";
   
   // We must EXPLICITLY specify StringLen(json) to avoid sending the hidden Null Terminator (\0)
   StringToCharArray(json, post, 0, StringLen(json), CP_UTF8);
   ResetLastError();
   
   int res = WebRequest("POST", bridge_url, req_headers, 500, post, result, result_headers);
   
   if(res == -1)
     {
      Print("🛑 MT5 FIREWALL BLOCKED IT. MT5 Error Code: ", GetLastError());
      Print("👉 You MUST add this exact domain to Tools -> Options -> Expert Advisors: https://discord.com");
     }
   else
      if(res == 200 || res == 204)
        {
         Print("✅ DISCORD SUCCESS: Message delivered beautifully.");
        }
      else
        {
         string error_body = CharArrayToString(result);
         Print("⚠️ DISCORD REJECTED IT. HTTP Status Code: ", res);
         Print("👉 Discord Server Reply: ", error_body);
        }
  }
